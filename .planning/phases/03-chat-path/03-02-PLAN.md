CREATE src/llm/llm.service.ts:
```typescript
import { Injectable, Logger, OnModuleInit } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { ChatAnthropic } from '@langchain/anthropic';
import type { BaseMessage } from '@langchain/core/messages';

@Injectable()
export class LlmService implements OnModuleInit {
  private readonly logger = new Logger(LlmService.name);
  private llm!: ChatAnthropic;

  constructor(private readonly config: ConfigService) {}

  onModuleInit(): void {
    const model = this.config.getOrThrow<string>('ANTHROPIC_MODEL');
    this.llm = new ChatAnthropic({ model, streaming: true });
    this.logger.log(`LlmService initialized with model=${model}`);
  }

  async *streamResponse(
    messages: BaseMessage[],
    signal: AbortSignal,
  ): AsyncIterable<string> {
    if (!this.llm) {
      throw new Error('[LlmService] llm not initialized — was onModuleInit called?');
    }
    const stream = await this.llm.stream(messages, { signal });
    for await (const chunk of stream) {
      const text =
        typeof chunk.content === 'string'
          ? chunk.content
          : chunk.content
              .filter(
                (c): c is { type: 'text'; text: string } => c.type === 'text',
              )
              .map((c) => c.text)
              .join('');
      if (text) yield text;
    }
  }
}
```

| AbortSignal | Provided by ChatGateway's per-socket AbortController; controls cancellation of the HTTP call to Anthropic |

| T-03-02-01 | Information Disclosure | LlmService — ANTHROPIC_MODEL env var | mitigate | ConfigService.getOrThrow('ANTHROPIC_MODEL') — never hardcoded; startup fails if env var is absent (ConfigModule.forRoot validation) |
| T-03-02-02 | Denial of Service | LlmService.streamResponse — unbounded stream | accept | AbortSignal is always passed from ChatGateway's AbortController; disconnect triggers abort; no server-side timeout added in v1 (Phase 6 hardening concern) |
| T-03-02-03 | Tampering | messages[] content from user | accept | LlmService does not inspect or validate message content — that is the gateway's responsibility; user content is wrapped in HumanMessage by gateway before reaching LlmService |
