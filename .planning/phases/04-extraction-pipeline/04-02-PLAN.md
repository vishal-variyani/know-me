    Create `src/extraction/nodes/classify.node.ts` as a pure factory function. No class, no DI decorators. The function is synchronous (D-01: no LLM, no async).

    ```typescript
    import { Logger } from '@nestjs/common';
    import type { ExtractionState } from '../extraction.types.js';

    // D-01: Trivial message patterns — must match entire trimmed content (case-insensitive)
    // These are single-word/short greetings and acknowledgements.
    // Multi-word messages that happen to contain these words are not trivial —
    // only exact single-word matches are filtered here. The regex anchors to full string.
    const TRIVIAL_PATTERN = /^(ok|okay|thanks|sure|yes|no|hi|hello|bye|yep|nope|cool|great|got it|sounds good)\.?!?$/i;

    // D-01: Must contain at least one proper noun (capitalized word not at sentence start)
    // Strategy: look for a capitalized word that is NOT the first word in the content.
    // This avoids false positives from sentence-initial capitalization.
    const PROPER_NOUN_PATTERN = /(?<!\.\s*|\n\s*|^\s*)(?<=[^.!?\n]\s)\b[A-Z][a-z]+\b|\b[A-Z][a-z]+\s+[A-Z][a-z]+\b/;

    export function makeClassifyNode(logger: Logger) {
      return function classifyNode(
        state: ExtractionState,
      ): Partial<ExtractionState> {
        const { content, correlationId } = state;
        const trimmed = content.trim();

        const isTrivial = TRIVIAL_PATTERN.test(trimmed);
        const hasProperNoun = PROPER_NOUN_PATTERN.test(trimmed);

        const shouldExtract = !isTrivial && hasProperNoun;

        logger.debug(
          `[${correlationId}] classifyNode shouldExtract=${shouldExtract} isTrivial=${isTrivial} hasProperNoun=${hasProperNoun} contentLen=${trimmed.length}`,
        );

        return { classifyResult: { shouldExtract } };
      };
    }
    ```

    IMPORTANT on the proper noun regex: The pattern above uses lookbehind to avoid matching the first word in a sentence. However, regex lookbehind with variable-length content has browser/node compat considerations. A simpler and more robust approach:

    1. Split the content into words.
    2. Skip the very first word.
    3. Check if any subsequent word matches `/^[A-Z][a-z]+$/`.

    Use this implementation instead (more readable and testable):

    ```typescript
    function hasProperNounInContent(content: string): boolean {
      const words = content.trim().split(/\s+/);
      // Check from index 1 to skip the sentence-opening capital
      for (let i = 1; i < words.length; i++) {
        const word = words[i].replace(/[^A-Za-z]/g, '');
        if (/^[A-Z][a-z]{1,}$/.test(word)) return true;
      }
      // Also check first word if the content starts with a name (e.g., "Jake is my friend")
      // by checking whether the WHOLE sentence is not trivial and first word is capitalized
      if (words.length >= 3 && words[0] && /^[A-Z][a-z]{1,}$/.test(words[0])) {
        return true;
      }
      return false;
    }
    ```

    Use `hasProperNounInContent(trimmed)` instead of the regex in the node function.

    **Node factory**:

    ```typescript
    const EMPTY_RESULT: ExtractOutput = {
      people: [],
      topics: [],
      emotionalTone: 'neutral',
      keyFacts: [],
    };

    export function makeExtractNode(llm: ChatOpenAI, logger: Logger) {
      const prompt = ChatPromptTemplate.fromMessages([
        ['system', EXTRACT_SYSTEM_PROMPT],
        ['human', '{content}'],
      ]);
      const chain = prompt.pipe(llm.withStructuredOutput(ExtractOutputSchema));

      return async function extractNode(
        state: ExtractionState,
      ): Promise<Partial<ExtractionState>> {
        const { content, correlationId } = state;

        // D-06: Try once, retry once on failure, absorb on second failure
        for (let attempt = 1; attempt <= 2; attempt++) {
          try {
            const result = await chain.invoke({ content });
            logger.debug(
              `[${correlationId}] extractNode people=${result.people.length} keyFacts=${result.keyFacts.length}`,
            );
            return { extractResult: result };
          } catch (err: unknown) {
            logger.error(
              `[${correlationId}] extractNode attempt=${attempt} failed: ${String(err)}`,
            );
            if (attempt === 2) {
              return { extractResult: EMPTY_RESULT };
            }
          }
        }
        // TypeScript exhaustiveness — unreachable
        return { extractResult: EMPTY_RESULT };
      };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LLM output (GPT-4o-mini) | JSON response crosses trust boundary — treated as untrusted input |
| User message content | Passed to LLM prompt — potential prompt injection vector |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-02-01 | Tampering | LLM JSON output → ExtractOutputSchema | mitigate | `llm.withStructuredOutput(ExtractOutputSchema)` uses Zod validation — malformed LLM output throws and triggers the D-06 retry-then-absorb path; invalid data never reaches Validate node |
| T-04-02-02 | Tampering | User content injected into prompt | accept | System prompt is a separate `['system', ...]` message; user content is in `['human', '{content}']` — LangChain message roles provide structural separation; no format-string interpolation |
| T-04-02-03 | Denial of Service | Classify node called on every BullMQ job | accept | Classify is synchronous and O(n) in content length; no external calls; not a DoS concern at BullMQ concurrency: 3 |
| T-04-02-04 | Information Disclosure | LLM error messages logged | mitigate | Errors logged via NestJS Logger at error level with correlationId; error strings from LangChain do not include user content in normal operation |
